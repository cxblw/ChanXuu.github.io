                          # @Prop

                          父子组件之间值的传递

                          <script lang="ts">
import { Vue, Component, Prop } from 'vue-property-decorator'
 
@Component
export default class YourComponent extends Vue {
  @Prop(number) readonly propA: number 
  @Prop([number,string]) readonly propB: number | string
  @Prop({type:Boolean,default:false}) readonly propC: boolean 
  @Prop({type:Object,default:()=>{ return {} }}) readonly propD: any
}
</script>

对应

<script>
export default {
  props: {
    propA: {
      type: Number,
    },
    propB: {
      type: [number,string],
    },
    propC: {
      type:Boolean,
      default:false
    },
    propD: {
      type:Object,
      default:()=>{ return {} }
    },
  },
}
</script>

                          # @PropSync

                          @PropSync 相当于在父组件中添加.sync装饰器, 使子组件也可以更新prop的值。

                          @PropSync 会生成一个新计算属性的 getter 和 setter.

                          父组件

                           <prop-sync-test :isShow.sync="isShow"></prop-sync-test>
<!--    <prop-sync-test :isShow="isShow" @update="val => isShow = val" ></prop-sync-test>    -->
                          1
                          2

                          子组件

                          <script lang="ts">
import { Vue, Component, PropSync } from 'vue-property-decorator'
@Component
export default class extends Vue {
  @PropSync('isShow', { type: String }) PropSyncName!: string; //用来实现组件的双向绑定，isShow是父组件传来的属性名
 
  changeName(): void {
    this.PropSyncName = '改变值'; // 更改PropSyncName会更改父组件的isShow的值
  }
}
</script>

                          # @Watch

                          监听事件

                          <script lang="ts">
import { Vue, Component, Watch } from 'vue-property-decorator'
 
@Component
export default class YourComponent extends Vue {
  @Watch('data', { immediate: true, deep: true })
  onChangeData(val: Person, oldVal: Person) {
  	//执行的代码
  }
}
</script>

对应的js写法
<script>
  watch: {
    data:{
      handler: 'onChangeData',
      immediate: false,
      deep: false,
    }
  },
  methods:{
    onChangeData(val, oldVal){}
  }
</script>

                          # @Provide / @Inject

                          单项数据流，不可以在后代组件直接修改祖先组件传过来的内容，直接修会报错，告知我们避免直接更改注入值,跟直接修改prop的性质一样，并且当祖先组件的值改变时，后代组件不响应数据变化（对象除外）

                          祖先组件

                          <template>
    <div>
        <hello1></hello1>
    </div>
</template>

<script lang="ts">
import { Component, Provide, Vue } from 'vue-property-decorator'
import hello1 from '@/components/hello1.vue'

@Component({
    name: 'ProvideInject',
    components: { hello1 }
})
export default class ProvideInject extends Vue {
@Provide('testList')
testList = [
    {id:1,name:'123'},
    {id:2,name:'1112'}
]
@Provide('getDate')
getDate = ()=>{
    return new Date()
}
}
</script>

                          # @ProvideReactive / @InjectReactive

                          用法和@Provide / @Inject的使用方法一样，但是这个是响应式的数据流，也就是说，当祖先组件改变了值，后代组件的值也响应更改

                          # @Emit

                          <script lang='ts'>import { Vue, Component, Emit } from 'vue-property-decorator' @Componentexport default class YourComponent extends Vue {  @Emit()  addToCount(n: number) {   }   @Emit('reset')  resetCount() {  } }</script>相当于<script> export default {  methods: {    addToCount(n) {       this.$emit('add-to-count', n)    },    resetCount() {       this.$emit('reset')    }   } }</script>

                          # @Ref

                          这种在写的时候 不去主动确定其类型（毕竟好多其实也不清楚是什么类型的）

                          @Ref() readonly refKey

                          如果我们确定其类型,也能写的具体些, 但是在使用属性的时候 容易报警 @Ref('refKey') readonly name!: HTMLButtonElement
