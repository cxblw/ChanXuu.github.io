# 百度地图添加谷歌瓦片

//创建百度地图
 var map = new BMap.Map("l-map",{mapType: BMAP_HYBRID_MAP,minZoom:1,maxZoom:17});
map.centerAndZoom(new BMap.Point(116.4035,39.915),12);
map.enableScrollWheelZoom();
var mk = new BMap.Marker(new BMap.Point(116.4035,39.915));
map.addOverlay(mk);
map.panTo(new BMap.Point(116.4035,39.915));
let baiduX = [0, 0, 1, 3, 6, 12, 24, 49, 98, 197, 395, 790, 1581, 3163, 6327, 12654, 25308, 50617 ];
let baiduY = [ 0, 0, 0, 1, 2, 4, 9, 18, 36, 73, 147, 294, 589, 1178, 2356, 4712, 9425, 18851 ];
let googleX = [0, 1, 3, 6, 13, 26, 52, 105, 210, 421, 843, 1685, 3372, 6744, 13489, 26979, 53958, 107917 ];
let googleY = [0, 0, 1, 2, 5, 12, 24, 48, 97, 194, 387, 776, 1551, 3103, 6207, 12415, 24830, 49661 ];
//获取对应的谷歌瓦片的x,y,zoom
var tilelayer = new BMap.TileLayer();
tilelayer.getTilesUrl=function(tileCoord, zoom) {
    var x = tileCoord.x;
    var y = tileCoord.y;
    // 谷歌瓦片行编号=[谷歌参照瓦片行编号+(百度行编号 – 百度参照瓦片行编号)]
    let gx = googleToBaiduX(x, zoom);
    let gy = googleToBaiduY(y, zoom);
    var url = 'http://www.google.cn/maps/vt?lyrs=y@189&gl=cn&x='+ gx + '&y=' + gy + '&z='+ zoom + '&s=Galil'
    //另一个瓦片地址:var url = "http://mt2.google.cn/vt/lyrs=s&scale=2&hl=zh-CN&gl=cn&x=" + x + "&y=" + y + "&z=" z ;
    // var url = 'images/tiles/' + zoom + '/tile' + gx + '_' + y + '.png';
    return url;
}
map.addTileLayer(tilelayer);
function googleToBaiduX(x, z) {
    let b = baiduX[z - 1];// 395
    let g = googleX[z];// 11:843,12:1685
    let gx = g + (x-b);// --- 1587+
    // let gx = x - g + b;// --- 1587+
    // 谷歌瓦片行编号=[谷歌参照瓦片行编号+(百度行编号 – 百度参照瓦片行编号)]
    return gx;
}

function googleToBaiduY(y, z) {
    let b = baiduY[z - 1];// 147
    let g = googleY[z];// 10:
    let gy = g - (y-b);//
    // let gy = g + b - y;//
    // 谷歌瓦片列编号=[谷歌参照瓦片列编号- (百度列编号 – 百度参照瓦片列编号)] //向上，列为递减
    return gy;
}

百度地图添加谷歌瓦片后,图像位置是有偏差的,原因是因为百度地图添加的谷歌瓦片就是有偏差,不是经纬度坐标系的偏差(已实验过,用谷歌地图上的坐标,在添加瓦片的百度地图上显示,有偏差,有百度地图的坐标,也是有偏差,而且缩放的层级越大,偏的越严重),实际原因暂时不知道

# 高德地图添加谷歌瓦片

//vue内写法
//生成高德地图
this.map = new AMap.Map("allmap", {
zoom: 16, //级别
zooms: [5, 18],
center: [107.78867208478735, 22.504817467539358], //中心点坐标
viewMode: "2D",
layers: [
    // 卫星
    // new AMap.TileLayer.Satellite(),

    new AMap.TileLayer.RoadNet()
]
});
this.map.setMapStyle('amap://styles/0880367c3dfd232148128c9c3049e181')
this.xyzTileLayer = new AMap.TileLayer({
// 图块取图地址
getTileUrl: function(x, y, z) {
    var url =
    "http://www.google.cn/maps/vt?lyrs=y@189&gl=cn&x=" +
    x +
    "&y=" +
    y +
    "&z=" +
    z +
    "&s=Galil";
    return url;
},
zIndex: 100
});
